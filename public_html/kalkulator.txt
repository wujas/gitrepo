function dodaj(){
  var liczba1 = document.getElementById("liczba1").value;
  var liczba2 = document.getElementById("liczba2").value;
  
  alert("Suma: " + (parseInt(liczba1)+parseInt(liczba2)));
  
}

function odejmij(){
  var liczba1 = document.getElementById("liczba1").value;
  var liczba2 = document.getElementById("liczba2").value;
  alert("Różnica: " + (parseInt(liczba1)-parseInt(liczba2)))
}

}
